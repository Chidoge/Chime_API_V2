This text file contains the queries for setting up the database tables

CREATE TABLE auth(username varchar(100) not null, hash varchar(100), lastSeen varchar(100));

CREATE TABLE profile(username varchar(100) not null, first varchar(100), last varchar(100), occupation varchar(100) default 'N/A', picture varchar(100) default 'http://puu.sh/BHa8O/c8e082dffe.png', about varchar(100) default 'N/A', birthday varchar(100) default 'N/A', location varchar(50) default 'N/A');

create table messages(sender varchar(100), destination varchar(100), message varchar(100), timestamp bigint, isImage smallint);
