This text file contains the queries for setting up the database tables

CREATE TABLE Login(id serial primary key, hash varchar(100), username text unique not null, lastSeen text);

CREATE TABLE Profile(id serial primary key, first varchar(100), last varchar(100), occupation varchar(30) default 'N/A', picture text default 'http://puu.sh/BHa8O/c8e082dffe.png', about varchar(100) default 'N/A', birthday text default 'N/A', location varchar(50) default 'N/A');

create table Messages(messageID serial primary key,sender smallint, destination smallint, message text, timestamp bigint, isImage smallint);
