This text file contains the queries for setting up the database tables

CREATE TABLE auth(username varchar(max) not null, hash varchar(max), lastSeen varchar(max));

CREATE TABLE profile(username varchar(max) not null, first varchar(100), last varchar(100), occupation varchar(max) default 'N/A', picture varchar(max) default 'http://puu.sh/BHa8O/c8e082dffe.png', about varchar(100) default 'N/A', birthday varchar(max) default 'N/A', location varchar(50) default 'N/A');

create table messages(sender varchar(max), destination varchar(max), message varchar(max), timestamp bigint, isImage smallint);
